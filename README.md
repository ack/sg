# vars file

    ---
    sg_api_base: https://hostname
    sg_api_key: devel-api-key
    sg_partner_id: devel-partner-id
    name: Albert Choi
    email: albert.choi@ctl.io
    title: Engineer
    product: SGE-MYSQL


# invoke

`cd servergeneral-next`
`ansible-playbook -vvv -e @sgvars.yml playbook.yml`
